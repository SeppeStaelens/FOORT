142// Adaptive subdivision Mesh: starts with evenly spaced, square Mesh,
143// then decides to subdivide certain squares of pixels into smaller squares,
144// based on which pixels have a bigger "weight", which is defined as the maximum
145// "distance" (using the Diagnostic value distance) between the upper-left
146// vertex of the square with the other three vertices of the square.
147class SquareSubdivisionMesh : public Mesh
148{
149public:
150 // default constructor not possible
151 SquareSubdivisionMesh() = delete;
152 // Constructor must be called with arguments:
153 // - maxPixels: max. nr of pixels that can be integrated in TOTAL, over all iterations (if 0, then this is infinite,
154 // i.e. we keep integrating until all squares are maximally subdivided or have weight 0)
155 // - initialPixels: initial number of pixels to integrate (spaced equally over the screen)
156 // - maxSubdivide: maximum number of times that we can subdivide squares (note: 1 denotes the
157 // initial grid, so 2 would be subdividing the squares once)
158 // - iterationPixels: maximum number of pixels to subdivide in each integration iteration
159 // (max number of pixels that will be integrates is then 5*iterationPixels)
160 // - initialSubToFinal: once we decide to subdivide a square, do we automatically keep subdividing it
161 // until we reach maxSubdivision?
162 // - valdiag: the "value" and "distance" Diagnostic to use
163 SquareSubdivisionMesh(largecounter maxPixels, largecounter initialPixels, int maxSubdivide, largecounter iterationPixels, bool initialSubToFinal,
164 DiagBitflag valdiag)
165 : m_InitialPixels{static_cast<pixelcoord>(sqrt(initialPixels)) * static_cast<pixelcoord>(sqrt(initialPixels))},
166 m_MaxSubdivide{maxSubdivide},
167 m_RowColumnSize{static_cast<pixelcoord>((sqrt(initialPixels) - 1) * ExpInt(2, maxSubdivide - 1) + 1)},
168 m_PixelsLeft{maxPixels}, m_MaxPixels{maxPixels}, m_InfinitePixels{maxPixels == 0}, m_IterationPixels{iterationPixels},
169 m_InitialSubDividideToFinal{initialSubToFinal}, Mesh(valdiag)
170 {
171 if constexpr (dimension != 4)
172 ScreenOutput("SquareSubdivisionMesh only defined in 4D!", OutputLevel::Level_0_WARNING);
173
174 // DEBUG message for constructor (can delete)
175 ScreenOutput("SquareSubdivisionMesh constructed: maxPixels: " + (m_InfinitePixels ? "infinite" : std::to_string(maxPixels)) + "; m_InitialPixels: " + std::to_string(m_InitialPixels) + "; m_RowColumnSize: " + std::to_string(m_RowColumnSize), OutputLevel::Level_4_DEBUG);
176
177 // Initialize the initial square, equally spaced grid
178 InitializeFirstGrid();
179 }
