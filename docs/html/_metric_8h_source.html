16// The abstract base class for all Metrics.
17class Metric
18{
19public:
20 // Virtual destructor to ensure correct destruction of descendants
21 virtual ~Metric() = default;
22
23 Metric(bool rlogscale = false);
24
25 // Basic functions that return the metric with indices down or up:
26 // pure virtual as they must be defined in the descendant classes.
27 //
28 // Get the metric at Point p, indices down
29 virtual TwoIndex getMetric_dd(const Point &p) const = 0;
30 // Get the metric at Point p, indices up
31 virtual TwoIndex getMetric_uu(const Point &p) const = 0;
32
33 // The following functions return the Christoffel and other derivative quantities of the metric.
34 // They are implemented for this base class, BUT are left as virtual functions to allow for
35 // other metrics to implement their own (more efficient)
36 // way of calculating them, if so desired.
37 //
38 // Get the Christoffel symbol, indices up-down-down
39 virtual ThreeIndex getChristoffel_udd(const Point &p) const;
40 // Get the Riemann tensor, indices up-down-down-down
41 virtual FourIndex getRiemann_uddd(const Point &p) const;
42 // Get the Kretschmann scalar
43 virtual real getKretschmann(const Point &p) const;
44
45 // Function to get the description of the metric
46 // (used for outputting to the screen while running and possibly to the output files)
47 // There is a base class implementation of this function returning an undescriptive string
48 virtual std::string getFullDescriptionStr() const;
49
50 bool getrLogScale() const;
51
52protected:
53 // The symmetries (coordinate Killing vectors) of the metric. Should be set by descendant constructor.
54 std::vector<int> m_Symmetries{};
55 // Are we using a logarithmic r coordinate?
56 const bool m_rLogScale;
57};
