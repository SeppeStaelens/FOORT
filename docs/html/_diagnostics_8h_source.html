64// Abstract base class for all diagnostics
65class Diagnostic
66{
67public:
68 // Constructor must initialize the pointer to its owner Geodesic
69 Diagnostic() = delete;
70 Diagnostic(Geodesic *const theGeodesic) : m_OwnerGeodesic{theGeodesic}
71 {
72 }
73
74 // Resets Diagnostic object. This is called when the owner Geodesic is reset in order to start integrating
75 // a new geodesic.
76 // The base class implementation only resets m_StepsSinceUpdated
77 // Descendants can override this if they need to reset additional internal variables
78 virtual void Reset();
79
80 // virtual destructor to ensure correct destruction of descendants
81 virtual ~Diagnostic() = default;
82
83 // This is the heart of the Diagnostic. It calls the helper function DecideUpdate(), and if this
84 // returns true, will update its internal status based on the current (new) state of its owner geodesic.
85 virtual void UpdateData() = 0;
86
87 // These functions are for use at the end of integration of a geodesic.
88 // getFullData() returns all the data stored in the Diagnostic as a string (for output to file)
89 virtual std::string getFullDataStr() const = 0;
90 // getFinalDataVal() associates a value to the geodesic corresponding to the final value of this diagnostic
91 // (used for determining coarseness of nearby geodesics)
92 virtual std::vector<real> getFinalDataVal() const = 0;
93
94 // Function used to determine distance between two values obtained from getFinalDataVal()
95 // (for determining coarseness of nearby geodesics)
96 // This should return a number >=0
97 virtual real FinalDataValDistance(const std::vector<real> &val1, const std::vector<real> &val2) const = 0;
98
99 // Getters for descriptions
100 // This returns the name (only) of the Diagnostic, as a string without spaces that will be appended
101 // to an output file (e.g. prefix_DiagName.ext). Must be implemented!
102 virtual std::string getNameStr() const = 0;
103 // This returns the full description of the Diagnostic. Default implementation
104 // returns GetDiagNameStr()
105 virtual std::string getFullDescriptionStr() const;
106
107protected:
108 // The geodesic that owns the Diagnostic (a const pointer to the Geodesic)
109 Geodesic *const m_OwnerGeodesic;
110
111 // Helper function to decide if the Diagnostic should indeed update its status, based on
112 // its UpdateFrequency struct information (which will come from the Diagnostics's DiagnosticOptions)
113 bool DecideUpdate(const UpdateFrequency &myUpdateFrequency);
114
115 // The diagnostic is itself in charge of keeping track of how many steps it has been since it has been updated
116 // The Diagnostic's DiagnosticOptions struct tells it how many steps it needs to wait between updates
117 largecounter m_StepsSinceUpdated{};
118};
