153template <class OutputType>
154bool ConfigReader::ConfigCollection::LookupValue(int SettingIndex, OutputType &theOutput) const
155{
156 // Get the setting value, if it is indeed of this type
157 const OutputType *OutputPointer{std::get_if<OutputType>(&m_Settings[SettingIndex].SettingValue)};
158 if (OutputPointer)
159 {
160 // Success, the setting is indeed of this type; set the output accordingly
161 theOutput = *OutputPointer;
162 return true;
163 }
164 return false;
165}
