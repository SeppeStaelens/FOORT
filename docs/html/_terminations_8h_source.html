48// Possible termination conditions that can be set by Terminations
49enum class Term
50{
51 Uninitialized = -1, // Geodesic has not been properly initialized yet with initial position/velocity
52 Continue = 0, // All is right, continue integrating geodesic
53 Horizon, // STOP, encountered horizon (set by HorizonTermination)
54 BoundarySphere, // STOP, encountered boundary sphere (set by BoundarySphereTermination)
55 TimeOut, // STOP, taken too many steps (set by TimeOutTermination)
56 ThetaSingularity, // STOP, too close to polar coordinate singularity (theta = 0 or theta = pi/2)
57 NaN, // STOP, NaN encountered in geodesic position or velocity
58 GeneralSingularity, // STOP, singularity encountered (of any codimension)
59
61 // Add a new Termination condition that your new Termination can set
62 // Sample code:

65 */
67
68 Maxterms // Number of termination conditions that exist

74// Abstract base class for all Terminations
75class Termination
76{
77public:
78 // Constructor must initialize the pointer to its owner Geodesic
79 Termination() = delete;
80 Termination(Geodesic *const theGeodesic) : m_OwnerGeodesic{theGeodesic}
81 {
82 }
83
84 // Resets Termination object. This is called when the owner Geodesic is reset in order to start integrating
85 // a new geodesic.
86 // The base class implementation only resets m_StepsSinceUpdated
87 // Descendants can override this if they need to reset additional internal variables
88 virtual void Reset();
89
90 // virtual destructor to ensure correct destruction of descendants
91 virtual ~Termination() = default;
92
93 // Function that is called to determine whether Termination wants to
94 // terminate the Geodesic. Returns Term::Continue if no termination wanted,
95 // otherwise it returns the appropriate Term condition
96 virtual Term CheckTermination() = 0;
97
98 // This returns the full description of the Termination
99 virtual std::string getFullDescriptionStr() const = 0;
100
101protected:
102 // The geodesic that owns the Termination (a const pointer to the Geodesic)
103 Geodesic *const m_OwnerGeodesic;
104
105 // Helper function to decide if the Termination should indeed update its status, based on
106 // UpdateNSteps (which is set to 0 if we always update)
107 bool DecideUpdate(largecounter UpdateNSteps);
108
109 // The termination is itself in charge of keeping track of how many steps it has been since it has been updated
110 // The Termination's TerminationOptions struct tells it how many steps it needs to wait between updates
111 largecounter m_StepsSinceUpdated{};
112};
